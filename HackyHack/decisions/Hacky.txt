country_decisions = {
	enable_colonization_hack = {
		potential = {
			num_of_colonists = 1
			NOT = { has_country_modifier = colonization_hack }
			# NOT = { monthly_income = 2 } # If you wouldn't even be able to afford a colonist...
		}
		allow = {
			num_of_colonists = 1
			stability = 0
		}
		effect = {
			add_country_modifier = {
				name = "colonization_hack"
				duration = -1
			}
		}
		ai_will_do = {
			factor = 0
		}
	}
	
	disable_colonization_hack = {
		potential = {
			has_country_modifier = colonization_hack
		}
		allow = {
			has_country_modifier = colonization_hack
			stability = 0
		}
		effect = {
			add_stability = -1 # Discourage unnecessary toggling
			remove_country_modifier = colonization_hack
		}
		ai_will_do = {
			factor = 0
		}
	}
}
