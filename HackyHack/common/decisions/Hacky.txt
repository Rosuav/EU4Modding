country_decisions = {
	enable_colonization_hack = {
		potential = {
			colonist = 1
		}
		allow = {
			colonist = 1
		}
		effect = {
			add_country_modifier = {
				name = "colonization_hack"
				duration = -1
			}
			set_country_flag = colonization_hack_activated
		}
		ai_will_do = {
			factor = 0
		}
	}
	
	disable_colonization_hack = {
		potential = {
			has_country_modifier = colonization_hack_activated
		}
		allow = {
			has_country_modifier = colonization_hack_activated
			stability = 0
		}
		effect = {
			add_stability = -1 # Discourage unnecessary toggling
			remove_country_modifier = colonization_hack
			clr_country_flag = colonization_hack_activated
		}
		ai_will_do = {
			factor = 0
		}
	}
}
