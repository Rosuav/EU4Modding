country_decisions = {
	reform_from_within = {
		potential = {
			religion = catholic
			ai = no
		}
		allow = {
			religion = catholic
			stability = 1
			reform_desire = 0.1
			years_of_income = 5
			years_of_manpower = 5
		}
		effect = {
			add_reform_desire_scale = -0.25
			add_reform_desire = -0.1
			add_years_of_income = -5
			add_yearly_manpower = -5
			add_stability = -4
			hidden_effect = {
				# Notification to self
				country_event = {
					id = reform_from_within.1
				}
				# Notification to everyone else. Neither of these has any effect, and they
				# differ only in the tone of their wording.
				every_known_country = {
					limit = { religion = catholic }
					country_event = {
						id = reform_from_within.2
					}
				}
				# The church will relapse after 25-50 years.
				# Note that, to completely prevent reform desire from growing,
				# you will need to take this decision twice, and both their
				# timers will be counting down.
				country_event = {
					id = reform_from_within.3
					days = 9125
					random = 9125
				}
			}
			custom_tooltip = unreform_later
		}
	}
}
